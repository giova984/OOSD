180  //
181  // Returns the semi-confidence interval
182  // (the mean vary in [mu - sci; mu + sci]) relative to prob (1 - a)
183  //
184  namespace {
185  // Functor to use with accumulate() in order
186  // to evaluate the variance:
187  // sigma^2 = Sum( (exper[i] - _mu)^2 ) / N*(N-1)
188  // being N the number of rounds
189  class V {
190  double _mu;
191  public:
192  V(double m) : _mu(m) {}
193  double operator()(double result, double value) const
194  {
195  return result + (value - _mu)*(value - _mu);
196  }
197  };
198  }

200  //
201  // Returns the variance for the experimental samples
202  //
203  double BaseStat::getVariance()
204  {
205  double sum = 0;
206  double mu; // the mean
207 
208  if (!_endOfSim) throw Exc(GET);
209  if (!_initFlag) throw Exc(NO_INIT);
210  if (_expNum < 3) throw Exc(NEED_3);
211 
212  mu = getMean();
213 
214  sum = accumulate(_exper.begin(), _exper.end(), 0.0, V(mu));
215  return sqrt(sum/((_expNum - 1) * _expNum));
216  }

218  double BaseStat::getConfInterval(CONFIDENCE_INTERVAL c)
219  {
220  double mu; // the mean
221  double s; // the variance
222 
223  if (!_endOfSim) throw Exc(GET);
224  if (!_initFlag) throw Exc(NO_INIT);
225  if (_expNum < 3) throw Exc(NEED_3);
226 
227  mu = getMean();
228  s = getVariance();
229  return t_student(c, (unsigned int) _expNum - 1) * s;
230  }
