44  void GenericVar::readPDF(ifstream &f, int mode)
45  {
46  int n;
47  double p;
48  double sum;
49 
50  sum = 0;
51  while(!f.eof()) {
52  f >> n;
53  f >> p;
54 #ifdef __DEBUG__
55  cout << n << "\t" << p << "\n";
56 #endif
57  if (!f.eof()) {
58  if (_pdf[n] != 0) {
59  string errMsg = Exc::_WRONGPDF + string("\n");
60  throw Exc(errMsg, "GenericVar");
61  }
62  sum += p;
63  _pdf[n] = p;
64  }
65  }
66 
67  if (sum > 1) {
68  string errMsg = Exc::_WRONGPDF + string("\n");
69  throw Exc(errMsg, "GenericVar");
70  }
71 
72  if (sum < (1.0 - PDF_ERR)) {
73  cerr << "Warning: PDF values sum to " << sum << " < 1\n";
74  if (mode == 0) {
75  _pdf[n] += (1 - sum);
76  } else {
77  _pdf[1] += (1 - sum);
78  }
79  }
80  }
