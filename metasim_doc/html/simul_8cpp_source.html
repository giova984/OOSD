55  // this function performs one single simulation step
56  // It returns the tick after the simulation step has been completed
57  const Tick Simulation::sim_step()
58  {
59  Event *temp;
60  Tick mytime;
61 
62  DBGENTER(_SIMUL_DBG_LEV);
63 
64  temp = Event::getFirst(); // takes the first event in the queue ...
65  if (temp == NULL) throw NoMoreEventsInQueue();
66  temp->drop(); // ... and extract it!
67 
68  mytime = temp->getTime(); // stores the current time
69 
70  DBGPRINT_3("Executing event action at time [", mytime, "]: ");
71 #ifdef __DEBUG__
72  temp->print();
73  print();
74 #endif
75 
76  setTime(mytime);
77 
78  temp->action(); // do what it is supposed to do...
79  if (temp->isDisposable()) // if it has to be deleted...
80  delete temp; // delete it!
81 
82  return mytime;
83  }

145  // Main function:
146  // This is the simulation engine
147  void Simulation::run(Tick endTick, int nRuns)
148  {
149  DBGENTER(_SIMUL_DBG_LEV);
150  bool initializeRuns = true;
151  bool terminateSim = true;
152 
153  if (nRuns < -1) {
154  cout << "Initialize stats" << endl;
155  initializeRuns = true;
156  terminateSim = false;
157  numRuns = 1;
158  nRuns = -nRuns;
159  }
160  else if (nRuns == -1) {
161  cout << "Will not initialize stats" << endl;
162  initializeRuns = false;
163  terminateSim = false;
164  numRuns = 1;
165  }
166  else if (nRuns == 0) {
167  cout << "Last Sim in the batch" << endl;
168  initializeRuns = false;
169  terminateSim = true;
170  numRuns = 1;
171  }
172  else if (nRuns == 1) {
173  cout << "One single run" << endl;
174  initializeRuns = true;
175  terminateSim = true;
176  numRuns = 1;
177  }
178  else numRuns = nRuns;
179 
180  if (numRuns == 2) {
181  cout << "Warning: Simulation cannot be "
182  "initialized with 2 runs" << endl;
183  cout << " Executing 3 runs!" << endl;
184  numRuns = 3;
185  }
186 
187  if (initializeRuns) initRuns(numRuns);
188 
189  // Ok, now starts the main cycle of the simulation.
190  // remember that actRuns is the actual run number
191  // while numRuns is the maximum number of runs.
192  actRuns = 0;
193  while (actRuns < numRuns) {
194  cout << "\n Run #" << actRuns << endl;
195 
196  initSingleRun();
197 
198  // MAIN CYCLE!!
199  try {
200  while (globTime < endTick) {
201  globTime = sim_step();
202  }
203  } catch (NoMoreEventsInQueue &e) {
204  cerr << "No more events in queue: simulation time ="
205  << globTime << endl;
206  }
207 
208  endSingleRun();
209 
210  actRuns++; // next run....
211  }
212  end = true;
213  if (terminateSim) endSim(); // the simulation is over!!
214  }
