56  bool Event::Cmp::operator() (Event* e1, Event* e2) const {
57  Tick firstTime, secondTime;
58 
59  firstTime = e1->getTime();
60  secondTime = e2->getTime();
61 
62  if (firstTime < secondTime) return true;
63  else if (firstTime == secondTime) {
64  if (e1->getPriority() < e2->getPriority()) return true;
65  else if (e1->getPriority() == e2->getPriority()) {
66  if (e1->_order < (unsigned long)e2->_order)
67  return true;
68  else return false;
69  }
70  else return false;
71  }
72  else return false;
